mymax <- function(x,y){
  num.max <- x
  if(y>x){
    num.max <-y
  }
  return(num.max)
}

pn <-function(i){
  check=0
  for (j in 1:i) {
    if(i%%j==0){
      check = check+1
    }
  }
  if(check==2){
    return(T)
  }
  else{
    return(F)
  }
}
